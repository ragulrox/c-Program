#include<stdio.h>
int main()
{
    
    int n,arr[3],min,max,i;
    printf("\n Enter the number=");
    scanf("\n%d",&n);
    for(i=0;i<n;i++)
    {
        printf("\n Enter element=%d",i);
        scanf("%d",&arr[i]);
    }
    min=arr[0];
    max=arr[0];
    for(i=1;i<n;i++)
    {
        if(arr[i]>max)
        {
            max=arr[i];
        }
        if(arr[i]<min)
        {
            min=arr[i];
        }
        }
        printf("\n max=%d",max);
        printf("\n min=%d",min);
    return 0;
}
